### Phase 2: Parser Enhancement (2-3 hours)

#### 2.1 Add Worker-ID Format Support

**Problem:** Logs have format `[0-1] Error in "Suite [TC_ID] Name"`

**Solution:** Add third parser mode for worker-prefixed logs

**Implementation:**
```javascript
// File: scripts/parsing/parser.js

/**
 * Worker-ID parser: Handles [Worker-ID] Error in "..." format
 * Common in multi-worker WDIO runs
 */
const extractFailuresFromLogWorkerFormat = (consoleText) => {
  const results = [];
  
  // Pattern: [0-1] Error in "Suite Name [TC_ID] Test Name"
  const ERROR_RE = /\[\d+-\d+\]\s+Error in "([^"]+)\s+\[((?:TC|QAC-|BCIN-|TSTR-|BUG-|TASK-)\d+[^\]]*)\]\s+([^"]+)"/gm;
  
  for (const match of consoleText.matchAll(ERROR_RE)) {
    const [fullMatch, suiteName, tcId, tcName] = match;
    
    // Find associated failure message and run label
    const startIdx = match.index;
    const nextMatchIdx = consoleText.indexOf('[0-', startIdx + fullMatch.length);
    const blockEnd = nextMatchIdx > 0 ? nextMatchIdx : startIdx + 2000;
    const errorBlock = consoleText.slice(startIdx, blockEnd);
    
    // Extract run label if present
    const runMatch = errorBlock.match(/[✗�]+\s+(run_\d+)/);
    const runLabel = runMatch ? runMatch[1] : 'run_1';
    
    // Extract failure details
    const failedMatch = errorBlock.match(/- Failed:(.+?)(?:\n|$)/);
    const failureMsg = failedMatch ? failedMatch[1].trim() : 'Unknown error';
    
    // Extract snapshot URL if present
    const urlMatch = errorBlock.match(/Visit\s+(http:\/\/[^:]+:3000\/[^\s]+)/);
    const snapshotUrl = urlMatch ? urlMatch[1] : null;
    
    // Extract file name from stack trace or spec path
    const fileMatch = errorBlock.match(/file:\/\/\/([^:]+\.spec\.js)/);
    const fileName = fileMatch ? fileMatch[1].split('/').pop() : 'unknown.spec.js';
    
    results.push({
      fileName,
      tcId,
      tcName: tcName.trim(),
      stepId: tcId,
      stepName: tcName.trim(),
      runLabel,
      failureType: failureMsg.includes('Screenshot') ? 'screenshot_mismatch' : 'generic_failure',
      failureMsg,
      fullErrorMsg: errorBlock.slice(0, 500), // First 500 chars
      snapshotUrl,
      retryCount: 1
    });
  }
  
  return results;
};

/**
 * Main parser with fallback chain
 */
const extractFailuresFromLog = (consoleText) => {
  // Try new V2 parser first (file-based)
  const fileBlocks = splitByFilePattern(consoleText);
  if (fileBlocks.length > 0) {
    // Process file-based logs...
    return extractFailuresFromLogV2(consoleText);
  }
  
  // Try worker-ID format parser
  const workerFailures = extractFailuresFromLogWorkerFormat(consoleText);
  if (workerFailures.length > 0) {
    return workerFailures;
  }
  
  // Fallback to legacy parser
  return extractFailuresFromLogLegacy(consoleText);
};
```

#### 2.2 Improve Failure Detection Heuristics

**Problem:** Some logs don't have clear TC ID markers

**Solution:** Extract from multiple sources:
1. Error message: `Error in "... [TC_ID] ..."`
2. Spec file name: `lockPageSizeAuthoringViewMode.spec.js` → infer TC from context
3. Test output JSON: Parse `.test-result/output.json` if available

**Implementation:**
```javascript
// File: scripts/parsing/extractors.js

/**
 * Extract TC info from error messages with flexible patterns
 */
const extractTestCaseInfoFromError = (errorLine) => {
  // Pattern 1: [Worker-ID] Error in "... [TC_ID] ..."
  const workerPattern = /Error in "([^"]*)\s+\[((?:TC|QAC-|BCIN-)\d+[^\]]*)\]\s+([^"]+)"/;
  let match = errorLine.match(workerPattern);
  if (match) {
    return { id: match[2], name: match[3].trim() };
  }
  
  // Pattern 2: Standard [TC_ID] Name
  const standardPattern = /\[((?:TC|QAC-|BCIN-)\d+[^\]]*)\]\s+([^\n:]+)/;
  match = errorLine.match(standardPattern);
  if (match) {
    return { id: match[1], name: match[2].trim() };
  }
  
  return null;
};

/**
 * Extract file name from full spec path or file:// URL
 */
const extractFileName = (text) => {
  // Pattern 1: file:///path/to/spec.js
  let match = text.match(/file:\/\/\/([^:]+\.spec\.js)/);
  if (match) {
    return match[1].split('/').pop();
  }
  
  // Pattern 2: specs/path/to/spec.js
  match = text.match(/specs\/([^:]+\.spec\.js)/);
  if (match) {
    return match[1].split('/').pop();
  }
  
  return 'unknown.spec.js';
};

module.exports = {
  extractTestCaseInfoFromError,
  extractFileName,
  // ... existing exports
};
```
